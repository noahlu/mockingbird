写了一个统计Google搜索前十名页面信息的脚本

2012-09-30 11:04:19 评论

有时候你可能想知道某些关键词在google搜索结果中前10名的页面PR和Alexa信息，如果不考虑时间，你完全可以把top10的页面挨个点击，然后去看他们各自页面的PR和Alexa排名，假如...你想分析的关键词有数十个，你这样做会把自己弄崩溃的。今天就写了个脚本来帮忙抓取这些信息： 
<script>
(function getLinkUrl(){
 var search = document.getElementById("search");
 var h3Lists = search.getElementsByTagName("h3");
 var linkUrl = "";
 for(var i=0; i<h3Lists.length;i++){
 var linkNode = h3Lists[i].firstChild;
 link = linkNode.href + "
";
 linkUrl += link; 
 }
 
 var pr = "PR
", alexa = "Alexa
";
 for(i=0;i<15;i++){
 var seoquake = document.getElementById("seoquake-row-"+i);
 if(!seoquake) continue;
 var tr1 = seoquake.firstChild.nextSibling;
 var link1 = tr1.lastChild;
 pr += link1.innerHTML + "
";
 var tr2 = tr1.nextSibling.nextSibling.nextSibling.nextSibling.nextSibling;
 var link2 = tr2.lastChild;
 alexa += link2.innerHTML + "
";
 }


 document.write(linkUrl);
 document.write(pr);
 document.write(alexa);
 document.close();
})();


</script>
在google中搜索一个关键词，用这个脚本就能自动收集top10页面的URL，PR和Alexa排名，整理后就成了这样： 使用这个脚本有一个前提是必须有SEOquake插件，并且打开了Google搜索结果的详情显示。当然，这个脚本临时写着凑合用的，用起来稍微有点不方便，以后有时间再把它优化下或者封装成FF插件什么的吧。
